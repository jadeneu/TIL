## 내가 짠 코드
```python
def solution(answers):
    answer = []
    p1 = [1,2,3,4,5]
    p2 = [2,1,2,3,2,4,2,5]
    p3 = [3,3,1,1,2,2,4,4,5,5]
    score = [0,0,0]
    
    for idx, val in enumerate(answers):
        if val == p1[idx%len(p1)]:
            score[0] += 1
        if val == p2[idx%len(p2)]:
            score[1] += 1
        if val == p3[idx%len(p3)]:
            score[2] += 1
            
    for idx, val in enumerate(score):
        if val == max(score):
            answer.append(idx+1)
            
    return answer
```
