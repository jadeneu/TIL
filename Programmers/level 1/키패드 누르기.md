```python
def solution(numbers, hand):
    answer = ''
    lpos = 10
    rpos = 12

    for num in numbers:
      if num in [1,4,7]:
        answer += 'L'
        lpos = num
      elif num in [3,6,9]:
        answer += 'R'
        rpos = num
      else:
        num = 11 if num == 0 else num

        absL = abs(num-lpos)
        absR = abs(num-rpos)

        if sum(divmod(absL,3)) > sum(divmod(absR,3)):
          answer += 'R'
          rpos = num
        elif sum(divmod(absL,3)) < sum(divmod(absR,3)):
          answer += 'L'
          lpos = num
        else:
          if hand == 'left':
            answer += 'L'
            lpos = num
          else:
            answer += 'R'
            rpos = num

    return answer
```
