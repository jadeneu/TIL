## 내가 짠 코드
```python
def solution(n, arr1, arr2):
    for i in range(n):
        arr1[i] = bin(arr1[i] | arr2[i])[2:]
        
        # 이진수가 n 보다 작다면 앞에 모자란 만큼 "0" 삽입
        if len(arr1[i]) < n:
            zero = n - len(arr1[i])
            tmp = ""
            for _ in range(zero):
                tmp += "0"
            arr1[i] = tmp + arr1[i]
    
    for idx,line in enumerate(arr1):
        tmp = ""
        for i in line:
            if i == '1':
                tmp += '#'
            else:
                tmp += ' '
        arr1[idx] = tmp
    
    return arr1
```
